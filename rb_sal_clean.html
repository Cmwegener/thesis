Richardson Bay Clean environment and read in data

Convert to date

rb$date<-as.POSIXct(rb$DateTimeStamp, format = c("%m/%d/%y %H:%M"))
rb$date<-as.Date(as.POSIXct(rb$date, format=c("%Y/%m/%d")))

Calculate weekly salinity average and sd

rb$week_year<-date2week(rb$date, floor_day=TRUE)
sal_week_av<-aggregate( Sal ~ week_year,rb, mean)
sal_week_sd<-aggregate(Sal~week_year,rb, sd)

Change column names

names(sal_week_av)[names(sal_week_av) == "Sal"] <- "week_sal_av"
names(sal_week_sd)[names(sal_week_sd) == "Sal"] <- "week_sal_sd"

Combine data frames

rb_sal_summary<-cbind(sal_week_av, sal_week_sd)

Add the monthly average and sd columns to the eos df

rb_new<-merge(rb, rb_sal_summary[, c("week_year", "week_sal_av", "week_sal_sd")], by="week_year")

Caluculate 3 standard deviations from monthly average and then remove salinity values that are greater.

rb_new$remove<-(3*rb_new$week_sal_sd)+rb_new$week_sal_av
removed_data<-filter(rb_new, Sal<=remove)
